{% block extra_js %}
(function() {
  let visibleRows = [];

  function setCopyStatus(msg, type) {
    const el = document.getElementById('copyStatus');
    el.textContent = msg;
    if (type) {
      showToast(msg, type);
    }
  }

  function formatLine(item) {
    const ts = item.created_at || '';
    const lvl = (item.level || '').padEnd(5);
    const comp = (item.component || '').padEnd(10);
    const msg = item.message || '';
    return `${ts} [${lvl}] ${comp} ${msg}`;
  }

  async function copyVisibleLogs() {
    const text = visibleRows.map(formatLine).join('\n');
    if (!text) {
      setCopyStatus('No logs to copy.', 'warning');
      return;
    }
    try {
      await navigator.clipboard.writeText(text);
      setCopyStatus('Copied to clipboard.', 'success');
    } catch (e) {
      setCopyStatus('Copy failed.', 'error');
    }
  }

  function downloadVisibleLogs() {
    const text = visibleRows.map(formatLine).join('\n');
    if (!text) {
      setCopyStatus('No logs to download.', 'warning');
      return;
    }
    const blob = new Blob([text + '\n'], { type: 'text/plain' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    const feed = (document.querySelector('input[name=feed]')?.value || 'all');
    a.download = `logs-${feed}-${new Date().toISOString().slice(0,19)}.log`;
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    URL.revokeObjectURL(url);
    setCopyStatus('Download started.', 'success');
  }

  async function refreshLogs() {
    const q = new URLSearchParams(new FormData(document.getElementById('filters'))).toString();
    try {
      const r = await fetch('/api/logs?' + q);
      if (!r.ok) throw new Error('HTTP ' + r.status);
      const d = await r.json();
      visibleRows = d.items || [];
      const lines = visibleRows.map(formatLine);
      document.getElementById('out').textContent = lines.length ? lines.join('\n') : 'No logs found for current filters.';
    } catch (err) {
      visibleRows = [];
      document.getElementById('out').textContent = 'Failed to load logs.';
      showToast('Failed to refresh logs: ' + err.message, 'error');
    }
  }

  document.getElementById('filters').addEventListener('submit', (e) => {
    e.preventDefault();
    showToast('Refreshing logs...', 'info', 1500);
    refreshLogs();
  });
  document.getElementById('copyBtn').addEventListener('click', copyVisibleLogs);
  document.getElementById('downloadBtn').addEventListener('click', downloadVisibleLogs);

  // Quick-filter buttons: set the level input and reload
  document.querySelectorAll('.level-filter').forEach((btn) => {
    btn.addEventListener('click', () => {
      document.querySelectorAll('.level-filter').forEach(b => b.classList.remove('active'));
      btn.classList.add('active');
      const lvl = btn.dataset.level || '';
      document.getElementById('levelInput').value = lvl;
      refreshLogs();
    });
  });

  setInterval(() => {
    if (document.getElementById('autorefresh').checked) refreshLogs();
  }, 5000);
  refreshLogs();
  showToast('Logs loaded. Auto-refresh enabled.', 'info', 2000);
})();
